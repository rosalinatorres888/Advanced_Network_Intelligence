    def preprocess_text(self, text):
        """Preprocess text for analysis"""
        if pd.isna(text):
            return []
        
        # Convert to lowercase
        text = text.lower()
        
        # Remove special characters and digits
        text = re.sub(r'[^a-zA-Z\s]', '', text)
        
        # Tokenize
        tokens = word_tokenize(text)
        
        # Remove stopwords and short words
        tokens = [token for token in tokens if token not in self.stopwords and len(token) > 2]
        
        # Lemmatize
        tokens = [self.lemmatizer.lemmatize(token) for token in tokens]
        
        return tokens
    
    def create_co_occurrence_matrix(self, window_size=5, min_freq=2):
        """Create co-occurrence matrix from articles text"""
        print("\n" + "="*70)
        print(" CREATING CO-OCCURRENCE MATRIX FROM ARTICLES")
        print("="*70)
        
        # Combine text columns
        combined_texts = []
        for _, row in self.articles_df.iterrows():
            article_text = ""
            for col in self.text_columns:
                if pd.notna(row[col]):
                    article_text += " " + str(row[col])
            combined_texts.append(article_text)
        
        # Preprocess all texts
        print(" Preprocessing texts...")
        processed_texts = [self.preprocess_text(text) for text in combined_texts]
        self.processed_texts = processed_texts
        
        # Create co-occurrence dictionary
        co_occurrence = defaultdict(lambda: defaultdict(int))
        
        print(" Building co-occurrence relationships...")
        for tokens in processed_texts:
            for i, word1 in enumerate(tokens):
                for j in range(max(0, i - window_size), min(len(tokens), i + window_size + 1)):
                    if i != j:
                        word2 = tokens[j]
                        co_occurrence[word1][word2] += 1
        
        # Get vocabulary with minimum frequency
        word_freq = Counter()
        for tokens in processed_texts:
            word_freq.update(tokens)
        
        vocab = [word for word, freq in word_freq.items() if freq >= min_freq]
        vocab = sorted(vocab)
        
        print(f" Vocabulary: {len(vocab)} words (min frequency: {min_freq})")
        
        # Create matrix
        matrix = np.zeros((len(vocab), len(vocab)))
        word_to_idx = {word: i for i, word in enumerate(vocab)}
        
        for word1 in vocab:
            for word2 in vocab:
                if word2 in co_occurrence[word1]:
                    i, j = word_to_idx[word1], word_to_idx[word2]
                    matrix[i][j] = co_occurrence[word1][word2]
        
        # Create DataFrame
        self.co_occurrence_matrix = pd.DataFrame(matrix, index=vocab, columns=vocab)
        
        print(f" Co-occurrence matrix created: {self.co_occurrence_matrix.shape}")
        print(f" Total co-occurrences: {matrix.sum():.0f}")
        
        return self.co_occurrence_matrix
    
    def build_network_from_articles(self, min_weight=2, top_connections=100):
        """Build network graph from co-occurrence matrix"""
        print("\n" + "="*70)
        print(" BUILDING NETWORK FROM ARTICLES")
        print("="*70)
        
        if self.co_occurrence_matrix is None:
            self.create_co_occurrence_matrix()
        
        # Initialize graph
        self.G_weighted = nx.Graph()
        
        # Add edges
        edges_added = 0
        all_weights = []
        
        for word1 in self.co_occurrence_matrix.index:
            for word2 in self.co_occurrence_matrix.columns:
                weight = self.co_occurrence_matrix.at[word1, word2]
                if pd.notna(weight) and weight >= min_weight and word1 != word2:
                    self.G_weighted.add_edge(word1, word2, weight=weight)
                    all_weights.append(weight)
                    edges_added += 1
        
        # Keep only top connections if too many
        if top_connections and edges_added > top_connections:
            edges_with_weights = [(u, v, d['weight']) for u, v, d in self.G_weighted.edges(data=True)]
            edges_with_weights.sort(key=lambda x: x[2], reverse=True)
            
            self.G_weighted = nx.Graph()
            for u, v, weight in edges_with_weights[:top_connections]:
                self.G_weighted.add_edge(u, v, weight=weight)
        
        # Remove isolated nodes
        isolated_nodes = list(nx.isolates(self.G_weighted))
        self.G_weighted.remove_nodes_from(isolated_nodes)
        
        print(f" Network built: {self.G_weighted.number_of_nodes()} nodes, {self.G_weighted.number_of_edges()} edges")
        
        return self.G_weighted
    
    def analyze_topics(self, n_topics=5):
        """Perform topic analysis on articles"""
        print("\n" + "="*70)
        print(" TOPIC ANALYSIS")
        print("="*70)
        
        # Combine texts
        combined_texts = []
        for _, row in self.articles_df.iterrows():
            article_text = ""
            for col in self.text_columns:
                if pd.notna(row[col]):
                    article_text += " " + str(row[col])
            combined_texts.append(article_text)
        
        # Vectorize texts
        vectorizer = CountVectorizer(max_features=100, stop_words='english', ngram_range=(1, 2))
        doc_term_matrix = vectorizer.fit_transform(combined_texts)
        
        # Perform LDA
        lda = LatentDirichletAllocation(n_components=n_topics, random_state=42)
        lda.fit(doc_term_matrix)
        
        # Get feature names
        feature_names = vectorizer.get_feature_names_out()
        
        print(f" Discovered {n_topics} topics:")
        topics = []
        for topic_idx, topic in enumerate(lda.components_):
            top_words = [feature_names[i] for i in topic.argsort()[-10:][::-1]]
            topics.append(top_words)
            print(f"\n Topic {topic_idx + 1}: {', '.join(top_words[:5])}")
        
        return topics
    
    def create_enhanced_network_visualization(self, figsize=(16, 12)):
        """Create enhanced network visualization"""
        print("\n" + "="*70)
        print(" CREATING ENHANCED NETWORK VISUALIZATION")
        print("="*70)
        
        if self.G_weighted is None:
            self.build_network_from_articles()
        
        # Calculate centrality measures
        centrality_measures = {
            'degree': nx.degree_centrality(self.G_weighted),
            'betweenness': nx.betweenness_centrality(self.G_weighted, weight='weight'),
            'pagerank': nx.pagerank(self.G_weighted, weight='weight')
        }
        
        # Detect communities
        try:
            import community as community_louvain
            communities = community_louvain.best_partition(self.G_weighted, weight='weight')
        except ImportError:
            communities = {}
            for i, component in enumerate(nx.connected_components(self.G_weighted)):
                for node in component:
                    communities[node] = i
        
        # Create layout
        pos = nx.spring_layout(self.G_weighted, k=2, iterations=50, weight='weight')
        
        # Set up plot
        fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=figsize, facecolor='white')
        fig.suptitle('Articles Network Analysis - Multiple Views', fontsize=20, fontweight='bold')
        
        # View 1: Community-based
        node_sizes_1 = [300 + centrality_measures['degree'][node] * 1500 for node in self.G_weighted.nodes()]
        node_colors_1 = [communities[node] for node in self.G_weighted.nodes()]
        
        nx.draw(self.G_weighted, pos, ax=ax1,
               node_size=node_sizes_1, node_color=node_colors_1,
               cmap=plt.cm.Set3, alpha=0.8, linewidths=2, edgecolors='white',
               edge_color='gray', width=0.5)
        ax1.set_title('Community Detection View', fontsize=14, fontweight='bold')
        
        # View 2: PageRank centrality
        node_sizes_2 = [300 + centrality_measures['pagerank'][node] * 2000 for node in self.G_weighted.nodes()]
        node_colors_2 = [centrality_measures['pagerank'][node] for node in self.G_weighted.nodes()]
        
        nx.draw(self.G_weighted, pos, ax=ax2,
               node_size=node_sizes_2, node_color=node_colors_2,
               cmap=plt.cm.viridis, alpha=0.8, linewidths=2, edgecolors='white',
               edge_color='lightgray', width=0.5)
        ax2.set_title('PageRank Centrality View', fontsize=14, fontweight='bold')
        
        # View 3: Strong connections only
        edges = list(self.G_weighted.edges(data=True))
        weights = [d['weight'] for u, v, d in edges]
        threshold = np.percentile(weights, 75)
        strong_edges = [(u, v) for u, v, d in edges if d['weight'] >= threshold]
        G_strong = self.G_weighted.edge_subgraph(strong_edges).copy()
        
        pos3 = nx.spring_layout(G_strong, k=2, iterations=50)
        nx.draw(G_strong, pos3, ax=ax3,
               node_size=800, node_color='lightcoral', alpha=0.8,
               linewidths=2, edgecolors='darkred', width=2,
               edge_color='darkred', with_labels=True, font_size=8, font_weight='bold')
        ax3.set_title('Strongest Connections (Top 25%)', fontsize=14, fontweight='bold')
        
        # View 4: Top keywords with labels
        top_nodes = sorted(centrality_measures['pagerank'].items(), key=lambda x: x[1], reverse=True)[:15]
        G_top = self.G_weighted.subgraph([node for node, _ in top_nodes]).copy()
        
        pos4 = nx.spring_layout(G_top, k=3, iterations=50)
        node_sizes_4 = [500 + centrality_measures['pagerank'][node] * 1500 for node in G_top.nodes()]
        
        nx.draw(G_top, pos4, ax=ax4,
               node_size=node_sizes_4, node_color='lightblue', alpha=0.8,
               linewidths=2, edgecolors='navy', width=1,
               edge_color='gray', with_labels=True, font_size=10, font_weight='bold')
        ax4.set_title('Top 15 Keywords Network', fontsize=14, fontweight='bold')
        
        # Remove axes for all subplots
        for ax in [ax1, ax2, ax3, ax4]:
            ax.axis('off')
        
        plt.tight_layout()
        return fig
